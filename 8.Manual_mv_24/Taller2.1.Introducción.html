Calculo de estad√≠sticos

datos.r = datos  %>%
  group_by(Estudiante, Sexo)  %>%   # Factores con los que cuenta la base
  summarise(prom.2 = mean(Variable_2), # Promedio de la variable  2
            de.2   = sd(Variable_2),   # Desviaci√≥n est√°ndar V.2 
            var.2  = var(Variable_2),  # Varianza de la variable 2
            n.2    = n(),              # Tama√±o de la muestra
            ee.2   = sd(Variable_2)/sqrt(n()),  # Error est√°ndar
            max.2  = max(Variable_2, na.rm=T), # Valor m√°ximo
            min.2  = min(Variable_2, na.rm=T)) # Valor m√≠nimo
