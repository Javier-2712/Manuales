Calculo de estad√≠sticos

group_by = ordena a los datos por los diferentes grupos de cada factor (Estudiante, Sexo).

Estad√≠sticos = mean(media), sd(desviaci√≥n est√°ndar), var(varianza), max(valor m√°ximo), min(valor m√≠nimo).

attach(datos)

# Base de datos con estad√≠sticos (datos.r)
datos.r = 
  datos  %>%
  group_by(Estudiante, Sexo)  %>%   # Factores con los que cuenta la base
  summarise(prom.2 = mean(Variable2), # Promedio de la variable  2
            de.2   = sd(Variable2),   # Desviaci√≥n est√°ndar V.2 
            var.2  = var(Variable2),  # Varianza de la variable 2
            n.2    = n(),              # Tama√±o de la muestra
            ee.2   = sd(Variable2)/sqrt(n()),  # Error est√°ndar
            max.2  = max(Variable2, na.rm=T), # Valor m√°ximo
            min.2  = min(Variable2, na.rm=T)) # Valor m√≠nimo
